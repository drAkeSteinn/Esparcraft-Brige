    40→/**
    41→ * Resultado de ejecución de un trigger
    42→ */
    43→export interface TriggerExecutionResult {
    44→  success: boolean;
    45→  data?: any;
    46→  error?: string;
    47→}
    48→
    49→/**
    50→ * ✅ FUNCIÓN CENTRAL DE EJECUCIÓN
    51→ * 
    52→ * Esta función hace EXACTAMENTE lo mismo que /api/reroute/route.ts
    53→ * pero sin la capa HTTP/NextResponse
    54→ * 
    55→ * Garantiza:
    56→ * ✅ Construcción de prompts idéntica
    57→ * ✅ Reemplazo de variables primarias (usando replaceVariables)
    58→ * ✅ Reemplazo de plantillas del Grimorio (usando resolveAllVariables)
    59→ * ✅ Misma lógica de LLM
    60→ * 
    61→ * @param payload - Payload completo del trigger
    62→ * @returns Resultado de la ejecución
    63→ */
    64→export async function executeTrigger(
    65→  payload: AnyTriggerPayload
    66→): Promise<TriggerExecutionResult> {
    67→  try {
    68→    const { mode } = payload;
    69→
    70→    // Verificar si resumen general está corriendo (solo para chat)
    71→    const systemConfig = await db.systemConfig.findUnique({
    72→      where: { key: 'resumen_general_status' }
    73→    });
    74→
    75→    if (systemConfig?.value === 'running' && mode === 'chat') {
    76→      return {
    77→        success: true,
    78→        data: { response: "resumen_general" }
    79→      };
    80→    }
    81→
    82→    // ✅ DELEGAR AL HANDLER CORRECTO
    83→    switch (mode) {
    84→      case 'chat':
    85→        return await executeChat(payload as ChatTriggerPayload);
    86→      
    87→      case 'resumen_sesion':
    88→        return await executeResumenSesion(payload as ResumenSesionTriggerPayload);
    89→      
    90→      case 'resumen_npc':
    91→        return await executeResumenNPC(payload as ResumenNPCTriggerPayload);
    92→      
    93→      case 'resumen_edificio':
    94→        return await executeResumenEdificio(payload as ResumenEdificioTriggerPayload);
    95→      
    96→      case 'resumen_pueblo':
    97→        return await executeResumenPueblo(payload as ResumenPuebloTriggerPayload);
    98→      
    99→      case 'resumen_mundo':
   100→        return await executeResumenMundo(payload as ResumenMundoTriggerPayload);
   101→      
   102→      case 'nuevo_lore':
   103→        return await executeNuevoLore(payload as NuevoLoreTriggerPayload);
   104→      
   105→      default:
   106→        return {
   107→          success: false,
   108→          error: `Unknown trigger mode: ${mode}`
   109→        };
   110→    }
   111→  } catch (error) {
   112→    console.error('[executeTrigger] Error:', error);
   113→    return {
   114→      success: false,
   115→      error: error instanceof Error ? error.message : 'Error desconocido'
   116→    };
   117→  }
   118→}

   123→
   124→async function executeResumenSesion(
   125→  payload: ResumenSesionTriggerPayload
   126→): Promise<TriggerExecutionResult> {
   127→  const { npcid, playersessionid, systemPrompt, lastSummary, chatHistory } = payload;
   128→
   129→  // Get NPC and session
   130→  const npc = await npcDbManager.getById(npcid);
   131→  if (!npc) {
   132→    return { success: false, error: `NPC with id ${npcid} not found` };
   133→  }
   134→
   135→  const session = await sessionDbManagerSingleton.getById(playersessionid);
   136→  if (!session) {
   137→    return { success: false, error: `Session ${playersessionid} not found` };
   138→  }
   139→
   140→  // ✅ LEER CONFIGURACIÓN DE minMessages (misma configuración que usa Resumen General)
   141→  let minMessages = 10; // Valor por defecto
   142→  try {
   143→    const fs = await import('fs/promises');
   144→    const path = await import('path');
   145→    const configPath = path.join(process.cwd(), 'db', 'resumen-general-config.json');
   146→    try {
   147→      const configContent = await fs.readFile(configPath, 'utf-8');
   148→      const config = JSON.parse(configContent);
   149→      minMessages = config.minMessages || 10;
   150→      console.log(`[executeResumenSesion] minMessages leído: ${minMessages}`);
   151→    } catch (error) {
   152→      console.log('[executeResumenSesion] No se pudo leer config de minMessages, usando valor por defecto: 10');
   153→    }
   154→  } catch (error) {
   155→    console.log('[executeResumenSesion] Error al leer config de minMessages, usando valor por defecto: 10');
   156→  }
   157→
   158→  // ✅ VERIFICAR QUE LA SESIÓN TENGA SUFICIENTES MENSAJES
   159→  if (session.messages.length < minMessages) {
   160→    console.log(`[executeResumenSesion] Sesión ${playersessionid} tiene ${session.messages.length} mensajes (< ${minMessages}), OMITIENDO resumen`);
   161→    return {
   162→      success: false,
   163→      error: `La sesión tiene solo ${session.messages.length} mensajes. Se requieren al menos ${minMessages} mensajes para generar el resumen.`
   164→    };
   165→  }
   166→
   167→  console.log(`[executeResumenSesion] Sesión ${playersessionid} tiene ${session.messages.length} mensajes (>= ${minMessages}), PROCEDIENDO con resumen`);
   168→
   169→  // Get context (world, pueblo, edificio)
   170→  const world = await worldDbManager.getById(npc.location.worldId);
   171→  const pueblo = npc.location.puebloId ? await puebloDbManager.getById(npc.location.puebloId) : undefined;
   172→  const edificio = npc.location.edificioId ? await edificioDbManager.getById(npc.location.edificioId) : undefined;
   173→
   174→  // ✅ LEER CONFIGURACIÓN DE SYSTEM PROMPT
   175→  let configSystemPrompt = systemPrompt;
   176→  if (!configSystemPrompt) {
   177→    try {
   178→      const fs = await import('fs/promises');
   179→      const path = await import('path');
   180→      const configPath = path.join(process.cwd(), 'db', 'resumen-sesion-trigger-config.json');
   181→      try {
   182→        const configContent = await fs.readFile(configPath, 'utf-8');
   183→        const config = JSON.parse(configContent);
   184→        configSystemPrompt = config.systemPrompt || '';
   185→      } catch (error) {
   186→        configSystemPrompt = '';
   187→      }
   188→    } catch (error) {
   189→      configSystemPrompt = '';
   190→    }
   191→  }
   192→
   193→  // ✅ CONSTRUIR EL PROMPT COMPLETO
   194→  // NOTA: NO usa plantillas del Grimorio (grimorioTemplates: [])
   195→  const basePrompt = buildCompleteSessionSummaryPrompt({
   196→    world,
   197→    pueblo,
   198→    edificio,
   199→    npc,
   200→    session
   201→  }, {
   202→    systemPrompt: configSystemPrompt,
   203→    lastSummary,
   204→    chatHistory: chatHistory || session.messages.map(m => `${m.role}: ${m.content}`).join('\n\n'),
   205→    grimorioTemplates: [] // ✅ NO USAR PLANTILLAS DE GRIMORIO
   206→  });
   207→
   208→  // ✅ REEMPLAZAR VARIABLES PRIMARIAS
   209→  const varContext: VariableContext = {
   210→    npc,
   211→    world,
   212→    pueblo,
   213→    edificio,
   214→    session,
   215→    char: getCardField(npc?.card, 'name', ''),
   216→    lastSummary
   217→  };
   218→  const resolvedPrompt = replaceVariables(basePrompt, varContext);
   219→
   220→  // ✅ CONSTRUIR MENSAJES
   221→  const messages: ChatMessage[] = [
   222→    {
   223→      role: 'system',
   224→      content: resolvedPrompt,
   225→      timestamp: new Date().toISOString()
   226→    }
   227→  ];
   228→
   229→  // ✅ LLAMAR AL LLM
   230→  const llmResponse = await callLLM(messages);
   231→
   232→  // ✅ GUARDAR EN DB
   233→  await sessionSummaryDbManager.create({
   234→    sessionId: playersessionid,
   235→    npcId: npcid,
   236→    playerName: session.playerName,
   237→    npcName: getCardField(npc.card, 'name', ''),
   238→    summary: llmResponse,
   239→    timestamp: new Date().toISOString(),
   240→    version: 1
   241→  });
   242→
   243→  // ✅ LIMPIAR MENSAJES DE LA SESIÓN DESPUÉS DE RESUMIR
   244→  await sessionDbManagerSingleton.clearMessages(playersessionid);
   245→  console.log(`[executeResumenSesion] Sesión ${playersessionid}: ${session.messages.length} mensajes eliminados después de resumir`);
   246→
   247→  return {
   248→    success: true,
   249→    data: { summary: llmResponse }
   250→  };
   251→}

   254→// RESUMEN DE NPC
   255→// ========================================================================
   256→
   257→async function executeResumenNPC(
   258→  payload: ResumenNPCTriggerPayload
   259→): Promise<TriggerExecutionResult> {
   260→  const { npcid, systemPrompt, allSummaries } = payload;
   261→
   262→  // Get NPC
   263→  const npc = await npcDbManager.getById(npcid);
   264→  if (!npc) {
   265→    return { success: false, error: `NPC with id ${npcid} not found` };
   266→  }
   267→
   268→  // Get context (world, pueblo, edificio)
   269→  const world = await worldDbManager.getById(npc.location.worldId);
   270→  const pueblo = npc.location.puebloId ? await puebloDbManager.getById(npc.location.puebloId) : undefined;
   271→  const edificio = npc.location.edificioId ? await edificioDbManager.getById(npc.location.edificioId) : undefined;
   272→
   273→  // ✅ LEER CONFIGURACIÓN
   274→  let configSystemPrompt = systemPrompt;
   275→  if (!configSystemPrompt) {
   276→    try {
   277→      const fs = await import('fs/promises');
   278→      const path = await import('path');
   279→      const configPath = path.join(process.cwd(), 'db', 'resumen-npc-trigger-config.json');
   280→      try {
   281→        const configContent = await fs.readFile(configPath, 'utf-8');
   282→        const config = JSON.parse(configContent);
   283→        configSystemPrompt = config.systemPrompt || '';
   284→      } catch (error) {
   285→        configSystemPrompt = '';
   286→      }
   287→    } catch (error) {
   288→      configSystemPrompt = '';
   289→    }
   290→  }
   291→
   292→  // ✅ OBTENER RESÚMENES DE SESIONES DEL NPC
   293→  const npcSummaries = await sessionSummaryDbManager.getByNPCId(npcid);
   294→  let formattedSummaries = allSummaries;
   295→
   296→  if (!formattedSummaries && npcSummaries.length > 0) {
   297→    const summariesByPlayer = npcSummaries.reduce((acc, s) => {
   298→      const playerName = s.playerName || 'Unknown';
   299→      if (!acc[playerName]) acc[playerName] = [];
   300→      acc[playerName].push(s);
   301→      return acc;
   302→    }, {} as Record<string, any[]>);
   303→
   304→    const memoriesSections: string[] = [];
   305→    for (const [playerName, summaries] of Object.entries(summariesByPlayer)) {
   306→      memoriesSections.push(`Memoria de ${playerName}`);
   307→      summaries.forEach(s => memoriesSections.push(s.summary));
   308→    }
   309→
   310→    formattedSummaries = `***
   311→MEMORIAS DE LOS AVENTUREROS
   312→${memoriesSections.join('\n')}
   313→***`;
   314→  }
   315→
   316→  // ✅ CALCULAR HASH ACTUAL DE LOS RESÚMENES DE SESIONES DEL NPC
   317→  const currentHash = generateSessionSummariesHash(npcSummaries);
   318→
   319→  // ✅ OBTENER ÚLTIMO RESUMEN GUARDADO DEL NPC
   320→  const npcSummaryMgr = new NPCSummaryManager();
   321→  const lastNPCSummary = await npcSummaryMgr.getLatest(npcid);
   322→
   323→  console.log(`[executeResumenNPC] NPC ${npcid} - Resúmenes: ${npcSummaries.length}, Hash actual: ${currentHash}`);
   324→
   325→  // ✅ VERIFICAR SI HUBO CAMBIOS
   326→  if (lastNPCSummary?.sessionHash === currentHash) {
   327→    console.log(`[executeResumenNPC] NPC ${npcid} - SIN CAMBIOS, SKIP`);
   328→    return {
   329→      success: false,
   330→      error: `No hubo cambios en las sesiones del NPC ${npcid}. Los resúmenes son iguales.`
   331→    };
   332→  }
   333→
   334→  console.log(`[executeResumenNPC] NPC ${npcid} - HAY CAMBIOS, PROCEDIENDO CON RESUMEN`);
   335→
   336→  // ✅ CONSTRUIR EL PROMPT COMPLETO
   337→  const existingMemory = {};
   338→  const npcName = getCardField(npc?.card, 'name', '');
   339→  const varContext: VariableContext = {
   340→    npc,
   341→    world,
   342→    pueblo,
   343→    edificio,
   344→    char: npcName
   345→  };
   346→
   347→  let messages = buildNPCSummaryPrompt(
   348→    npc,
   349→    [],
   350→    existingMemory,
   351→    {
   352→      systemPrompt: configSystemPrompt,
   353→      allSummaries: formattedSummaries
   354→    }
   355→  );
   356→
   357→  // ✅ REEMPLAZAR VARIABLES PRIMARIAS Y PLANTILLAS DE GRIMORIO
   358→  const grimorioCards = grimorioManager.getAll();
   359→  const systemPromptRaw = messages[0]?.content || '';
   360→  const { result: systemPromptResolved } = resolveAllVariables(
   361→    systemPromptRaw, 
   362→    varContext, 
   363→    grimorioCards
   364→  );
   365→
   366→  messages = [
   367→    {
   368→      role: 'system',
   369→      content: systemPromptResolved,
   370→      timestamp: new Date().toISOString()
   371→    },
   372→    ...messages.slice(1)
   373→  ];
   374→
   375→  // ✅ LLAMAR AL LLM
   376→  const llmResponse = await callLLM(messages);
   377→
   378→  // ✅ GUARDAR RESUMEN EN creator_notes DE LA CARD DEL NPC (EN LA DB)
   379→  if (npc?.card) {
   380→    const updatedCard: any = { ...npc.card };
   381→
   382→    if (updatedCard.data) {
   383→      updatedCard.data = {
   384→        ...updatedCard.data,
   385→        creator_notes: llmResponse
   386→      };
   387→    } else {
   388→      updatedCard.data = {
   389→        creator_notes: llmResponse
   390→      };
   391→    }
   392→
   393→    await npcDbManager.update(npcid, { card: updatedCard });
   394→    console.log(`[executeResumenNPC] Resumen guardado en creator_notes de la card del NPC ${npcid}`);
   395→  }
   396→
   397→  // ✅ GUARDAR RESUMEN ANTERIOR COMO EMBEDDING
   398→  try {
   399→    if (lastNPCSummary?.summary && lastNPCSummary.summary.trim()) {
   400→      const { namespaceManager, buildNamespace } = await import('./namespaceManager');
   401→      const { getEmbeddingClient } = await import('./embeddings/client');
   402→      const ns = buildNamespace('npc', npcid);
   403→      await namespaceManager.ensureNpcNamespace(npcid);
   404→      const embClient = getEmbeddingClient();
   405→      try { await embClient.deleteBySource('resumen_npc_anterior', npcid); } catch {}
   406→      await embClient.createEmbedding({
   407→        content: `Resumen anterior (v${lastNPCSummary.version}) del NPC:\n\n${lastNPCSummary.summary}`,
   408→        metadata: { title: `Resumen anterior - NPC ${npcid}`, type: 'resumen_npc_anterior', npcId: npcid, version: lastNPCSummary.version, timestamp: new Date().toISOString() },
   409→        namespace: ns, source_type: 'resumen_npc_anterior', source_id: npcid,
   410→      });
   411→      console.log(`[executeResumenNPC] Resumen anterior guardado como embedding en "${ns}"`);
   412→    }
   413→  } catch (e: any) { console.error(`[executeResumenNPC] Error guardando embedding:`, e?.message); }
   414→
   415→  // ✅ GUARDAR EN TABLA NPCSummary PARA HISTÓRICO (CON VERSIÓN)
   416→  const nextVersion = (lastNPCSummary?.version || 0) + 1;
   417→  await npcSummaryMgr.create({
   418→    npcId: npcid,
   419→    summary: llmResponse,
   420→    sessionHash: currentHash,
   421→    version: nextVersion
   422→  });
   423→  console.log(`[executeResumenNPC] NPC ${npcid} - Resumen guardado en DB con versión ${nextVersion}`);
   424→
   425→  return {
   426→    success: true,
   427→    data: { summary: llmResponse, version: nextVersion }
   428→  };
   429→}

   443→  }
   444→
   445→  // Get context (world, pueblo)
   446→  const world = await worldDbManager.getById(edificio.worldId);
   447→  const pueblo = edificio.puebloId ? await puebloDbManager.getById(edificio.puebloId) : undefined;
   448→
   449→  // ✅ LEER CONFIGURACIÓN
   450→  let configSystemPrompt = systemPrompt;

   457→        const configContent = await fs.readFile(configPath, 'utf-8');

   459→        configSystemPrompt = config.systemPrompt || '';
   460→      } catch (error) {
   461→        configSystemPrompt = '';
   462→      }
   463→    } catch (error) {
   464→      configSystemPrompt = '';
   465→    }
   466→  }

   469→  const npcs = await npcDbManager.getByEdificioId(edificioid);

   475→        npcName: npc.card?.data?.name || npc.card?.name || 'Unknown',

   477→      };
   478→    })
   479→    .filter(n => n.consolidatedSummary !== '');
   480→
   481→  // ✅ CALCULAR HASH DE LOS RESÚMENES DE NPCs DEL EDIFICIO
   482→  const npcSummaryMgr = new NPCSummaryManager();
   483→  const allNPCSummaries = [];
   484→
   485→  for (const npc of npcs) {
   486→    const npcSummaries = await npcSummaryMgr.getByNPCId(npc.id);

   488→      allNPCSummaries.push(npcSummaries);
   489→    }
   490→  }
   491→
   492→  const currentHash = generateNPCSummariesHash(allNPCSummaries);
   493→
   494→  // ✅ OBTENER ÚLTIMO RESUMEN GUARDADO DEL EDIFICIO
   495→  const edificioSummaryMgr = new EdificioSummaryManager();
   496→  const lastEdificioSummary = await edificioSummaryMgr.getLatest(edificioid);
   497→
   498→  console.log(`[executeResumenEdificio] Edificio ${edificioid} - NPCs: ${npcs.length}, Hash actual: ${currentHash}`);
   499→
   500→  // ✅ VERIFICAR SI HUBO CAMBIOS EN LOS RESÚMENES DE NPCs
   501→  if (lastEdificioSummary?.npcHash === currentHash) {
   502→    console.log(`[executeResumenEdificio] Edificio ${edificioid} - SIN CAMBIOS, SKIP`);
   503→    return {
   504→      success: false,
   505→      error: `No hubo cambios en los resúmenes de NPCs del edificio ${edificioid}. Los resúmenes son iguales.`
   506→    };
   507→  }
   508→
   509→  console.log(`[executeResumenEdificio] Edificio ${edificioid} - HAY CAMBIOS, PROCEDIENDO CON RESUMEN`);
   510→
   511→  // ✅ CONSTRUIR EL PROMPT COMPLETO
   512→  const edificioName = edificio.name;
   513→  const varContext: VariableContext = {
   514→    edificio,
   515→    world,
   516→    pueblo,
   517→    char: edificioName
   518→  };
   519→
   520→  let messages = buildEdificioSummaryPrompt(
   521→    edificio,
   522→    [],
   523→    undefined,
   524→    {
   525→      systemPrompt: configSystemPrompt
   526→    }
   527→  );
   528→
   529→  // ✅ REEMPLAZAR VARIABLES PRIMARIAS Y PLANTILLAS DE GRIMORIO
   530→  const grimorioCards = grimorioManager.getAll();
   531→  const systemPromptRaw = messages[0]?.content || '';
   532→  const { result: systemPromptResolved } = resolveAllVariables(
   533→    systemPromptRaw, 
   534→    varContext, 
   535→    grimorioCards
   536→  );
   537→
   538→  messages = [
   539→    {
   540→      role: 'system',
   541→      content: systemPromptResolved,
   542→      timestamp: new Date().toISOString()
   543→    },
   544→    ...messages.slice(1)
   545→  ];
   546→
   547→  // ✅ LLAMAR AL LLM
   548→  const llmResponse = await callLLM(messages);
   549→
   550→  // ✅ GUARDAR RESUMEN EN lore (Estado del Edificio) DEL EDIFICIO (EN LA DB)
   551→  await edificioDbManager.update(edificioid, {
   552→    lore: llmResponse

   619→        configSystemPrompt = '';
   620→      }
   621→    } catch (error) {
   622→      configSystemPrompt = '';
   623→    }
   624→  }
   625→
   626→  // ✅ OBTENER lore (Estado del Edificio) DE LOS EDIFICIOS
   627→  const edificios = await edificioDbManager.getByPuebloId(pueblid);

   629→    .map(edificio => {

   631→      return {
   632→        edificioId: edificio.id,
   633→        edificioName: edificio.name,
   634→        consolidatedSummary
   635→      };
   636→    })
   637→    .filter(e => e.consolidatedSummary !== '');
   638→
   639→  // ✅ CALCULAR HASH DE LOS RESÚMENES DE EDIFICIOS DEL PUEBLO
   640→  const edificioSummaryMgr = new EdificioSummaryManager();
   641→  const allEdificioSummaries = [];
   642→
   643→  for (const edificio of edificios) {
   644→    const edificioSummaries = await edificioSummaryMgr.getByEdificioId(edificio.id);
   645→    if (edificioSummaries) {
   646→      allEdificioSummaries.push(edificioSummaries);
   647→    }
   648→  }
   649→
   650→  const currentHash = generateEdificioSummariesHash(allEdificioSummaries);
   651→

   655→
   656→  console.log(`[executeResumenPueblo] Pueblo ${pueblid} - Edificios: ${edificios.length}, Hash actual: ${currentHash}`);
   657→

   659→  if (lastPuebloSummary?.edificioHash === currentHash) {
   660→    console.log(`[executeResumenPueblo] Pueblo ${pueblid} - SIN CAMBIOS, SKIP`);

   662→      success: false,
   663→      error: `No hubo cambios en los resúmenes de edificios del pueblo ${pueblid}. Los resúmenes son iguales.`
   664→    };
   665→  }
   666→
   667→  console.log(`[executeResumenPueblo] Pueblo ${pueblid} - HAY CAMBIOS, PROCEDIENDO CON RESUMEN`);
   668→

   672→    pueblo,
   673→    world,
   674→    char: puebloName

   677→  let messages = buildPuebloSummaryPrompt(
   678→    pueblo,
   679→    [],
   680→    undefined,

   682→      systemPrompt: configSystemPrompt
   683→    }

   697→      role: 'system',

   714→  console.log(`[executeResumenPueblo] Resumen guardado en lore.eventos del pueblo ${pueblid}`);
