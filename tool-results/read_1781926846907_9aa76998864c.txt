    77→// Genera un ID único con fallback
    78→function makeId(): string {
    79→  try {
    80→    if (typeof crypto !== 'undefined' && typeof crypto.randomUUID === 'function') {
    81→      return crypto.randomUUID();
    82→    }
    83→  } catch {
    84→    /* ignore */
    85→  }
    86→  return `${Date.now()}-${Math.random().toString(36).slice(2, 11)}`;
    87→}

   480→    } catch (error) {

   482→      toast({
   483→        title: 'Error',
   484→        description: 'No se pudo guardar la plantilla',
   485→        variant: 'destructive'
