    61→/**
    62→ * Extrae JSON de una respuesta del LLM
    63→ * Maneja diferentes formatos: JSON puro, markdown con ```json, texto con JSON embebido
    64→ */
    65→function extractJsonFromResponse(response: string): string | null {
    66→  // Intentar parsear directamente
    67→  const trimmed = response.trim();
    68→  if (trimmed.startsWith('{') || trimmed.startsWith('[')) {
    69→    return trimmed;
    70→  }
    71→
    72→  // Buscar JSON en bloques markdown ```json ... ```
    73→  const jsonBlockMatch = response.match(/```(?:json)?\s*([\s\S]*?)```/);
    74→  if (jsonBlockMatch) {
    75→    return jsonBlockMatch[1].trim();
    76→  }
    77→
    78→  // Buscar JSON embebido (primer { hasta último })
    79→  const firstBrace = response.indexOf('{');
    80→  const lastBrace = response.lastIndexOf('}');
    81→  if (firstBrace !== -1 && lastBrace !== -1 && lastBrace > firstBrace) {
    82→    return response.substring(firstBrace, lastBrace + 1);
    83→  }
    84→
    85→  return null;
    86→}

    88→/**
    89→ * Intenta parsear JSON y validar que tenga las keys requeridas
    90→ */
    91→function tryParseJson(jsonStr: string, requiredKeys?: string[]): { success: boolean; data?: Record<string, any>; error?: string } {
    92→  try {
    93→    const parsed = JSON.parse(jsonStr);
    94→    
    95→    // Si hay keys requeridas, validar que existan
    96→    if (requiredKeys && Array.isArray(requiredKeys)) {
    97→      const missingKeys = requiredKeys.filter(key => !(key in parsed));
    98→      if (missingKeys.length > 0) {
    99→        return { success: false, error: `Missing required keys: ${missingKeys.join(', ')}` };
   100→      }
   101→    }
   102→    
   103→    return { success: true, data: parsed };
   104→  } catch (error) {
   105→    return { success: false, error: error instanceof Error ? error.message : 'Unknown parse error' };
   106→  }
   107→}

   118→/**
   119→ * Procesa la respuesta del LLM en modo JSON
   120→ * Intenta extraer, validar y corregir si es necesario
   121→ */
   122→async function processJsonResponse(
   123→  response: string,
   124→  jsonConfig: JsonResponseConfig,
   125→  messages: ChatMessage[]
   126→): Promise<JsonProcessResult> {
   127→  const metadata = {
   128→    jsonMode: true,
   129→    attempts: 1,
   130→    corrected: false,
   131→    usedFallback: false
   132→  };
   133→
   134→  // Intentar extraer JSON
   135→  let jsonStr = extractJsonFromResponse(response);
   136→  
   137→  if (!jsonStr) {
   138→    console.log('[processJsonResponse] No se pudo extraer JSON de la respuesta');
   139→    // Intentar corregir con LLM
   140→    return await attemptCorrection(response, jsonConfig, messages, metadata, 'No se encontró JSON en la respuesta');
   141→  }
   142→
   143→  // Intentar parsear
   144→  let parseResult = tryParseJson(jsonStr);
   145→  
   146→  if (parseResult.success) {
   147→    console.log('[processJsonResponse] JSON parseado exitosamente en primer intento');
   148→    return {
   149→      success: true,
   150→      data: parseResult.data!,
   151→      rawResponse: response,
   152→      metadata
   153→    };
   154→  }
   155→
   156→  console.log('[processJsonResponse] Error parseando JSON:', parseResult.error);
   157→  
   158→  // Intentar corregir
   159→  return await attemptCorrection(response, jsonConfig, messages, metadata, parseResult.error || 'Parse error');
   160→}

   162→/**
   163→ * Intenta corregir la respuesta usando el LLM

   165→async function attemptCorrection(
   166→  originalResponse: string,
   167→  jsonConfig: JsonResponseConfig,
   168→  originalMessages: ChatMessage[],
   169→  metadata: any,
   170→  errorMessage: string
   171→): Promise<JsonProcessResult> {
   172→  const maxRetries = jsonConfig.maxRetries || 2;
   173→  
   174→  for (let attempt = 1; attempt <= maxRetries; attempt++) {
   175→    metadata.attempts = attempt + 1;
   176→    console.log(`[attemptCorrection] Intento de corrección ${attempt}/${maxRetries}`);
   177→    
   178→    // Crear mensaje de corrección
   179→    const correctionPrompt = jsonConfig.correctionPrompt || 
   180→      `La respuesta anterior no tiene el formato JSON correcto. Error: ${errorMessage}
   181→
   182→Por favor, corrige la respuesta y devuelve ÚNICAMENTE el JSON válido sin texto adicional.
   183→
   184→Respuesta incorrecta:
   185→${originalResponse}
   186→
   187→Formato esperado:
   188→${JSON.stringify(jsonConfig.exampleResponse || jsonConfig.schema, null, 2)}`;
   189→
   190→    const correctionMessages: ChatMessage[] = [
   191→      ...originalMessages,
   192→      {
   193→        role: 'assistant',
   194→        content: originalResponse,
   195→        timestamp: new Date().toISOString()
   196→      },
   197→      {
   198→        role: 'user',
   199→        content: correctionPrompt,
   200→        timestamp: new Date().toISOString()
   201→      }
   202→    ];
   203→
   204→    try {
   205→      const correctedResponse = await callLLM(correctionMessages);
   206→      console.log('[attemptCorrection] Respuesta de corrección recibida');
   207→      
   208→      const jsonStr = extractJsonFromResponse(correctedResponse);
   209→      if (!jsonStr) {
   210→        console.log('[attemptCorrection] Aún no se puede extraer JSON');
   211→        originalResponse = correctedResponse;
   212→        continue;
   213→      }
   214→
   215→      const parseResult = tryParseJson(jsonStr);
   216→      if (parseResult.success) {
   217→        console.log('[attemptCorrection] JSON corregido exitosamente');
   218→        metadata.corrected = true;
   219→        return {
   220→          success: true,
   221→          data: parseResult.data!,
   222→          rawResponse: correctedResponse,
   223→          metadata
   224→        };
   225→      }
   226→      
   227→      originalResponse = correctedResponse;
   228→      errorMessage = parseResult.error || 'Parse error';
   229→    } catch (error) {
   230→      console.error('[attemptCorrection] Error llamando al LLM:', error);
   231→    }
   232→  }
   233→
   234→  // Agotados los intentos, usar fallback si existe
   235→  if (jsonConfig.fallbackResponse) {
   236→    console.log('[attemptCorrection] Usando respuesta de fallback');
   237→    metadata.usedFallback = true;
   238→    return {
   239→      success: true,
   240→      data: jsonConfig.fallbackResponse,
   241→      rawResponse: originalResponse,
   242→      metadata
   243→    };
   244→  }
   245→
   246→  // Sin fallback, retornar error
   247→  return {
   248→    success: false,
   249→    data: originalResponse,
   250→    rawResponse: originalResponse,
   251→    metadata: {
   252→      ...metadata,
   253→      error: `Failed after ${maxRetries} correction attempts: ${errorMessage}`
   254→    }
   255→  };
   256→}

   258→async function callLLM(messages: ChatMessage[]): Promise<string> {
   259→  try {
   260→    const response = await fetch(LLM_API_URL, {
   261→      method: 'POST',
   262→      headers: {
   263→        'Content-Type': 'application/json'
   264→      },
   265→      body: JSON.stringify({
   266→        model: LLM_MODEL,
   267→        messages: messages.map(m => ({ role: m.role, content: m.content })),
   268→        temperature: LLM_TEMPERATURE,
   269→        max_tokens: LLM_MAX_TOKENS
   270→      })
   271→    });
   272→
   273→    if (!response.ok) {
   274→      throw new Error(`LLM API error: ${response.status} ${response.statusText}`);
   275→    }
   276→
   277→    const data = await response.json();
   278→    return data.choices[0]?.message?.content || '';
   279→  } catch (error) {
   280→    console.error('Error calling LLM:', error);
   281→    throw error;
   282→  }
   283→}

   285→/**
   286→ * Realiza un merge incremental de datos del jugador
   287→ * - Los nuevos datos sobrescriben los existentes
   288→ * - Los datos que no vienen se conservan
   289→ * - Los campos vacíos en el payload NO borran los existentes (solo null borra)
   290→ */
   291→function mergeJugadorData(
   292→  jugadorExistente: Jugador | undefined,
   293→  jugadorNuevo: Jugador | undefined
   294→): Jugador | undefined {
   295→  // Si no hay datos nuevos, conservar existentes
   296→  if (!jugadorNuevo || Object.keys(jugadorNuevo).length === 0) {
   297→    return jugadorExistente;
   298→  }
   299→
   300→  // Si no hay datos existentes, usar nuevos (filtrando vacíos)
   301→  if (!jugadorExistente) {
   302→    const jugadorFiltrado = Object.entries(jugadorNuevo)
   303→      .filter(([_, valor]) => valor !== undefined && valor !== '')
   304→      .reduce((obj, [key, valor]) => ({ ...obj, [key]: valor }), {});
   305→
   306→    return Object.keys(jugadorFiltrado).length > 0 ? jugadorFiltrado : undefined;
   307→  }
   308→
   309→  // Merge: nuevos sobrescriben existentes
   310→  // Campos vacíos en payload NO borran, conservan valor anterior
   311→  // null en payload SÍ borra explícitamente
   312→  const merged = { ...jugadorExistente };
   313→
   314→  for (const [key, valor] of Object.entries(jugadorNuevo)) {
   315→    if (valor === null) {
   316→      // null significa borrar explícitamente
   317→      delete (merged as any)[key];
   318→    } else if (valor !== undefined && valor !== '') {
   319→      // Solo actualizar si el nuevo valor no es vacío
   320→      (merged as any)[key] = valor;
   321→    }
   322→    // Si valor es "" o undefined, conservar el existente
   323→  }
   324→
   325→  return merged;
   326→}

   499→  if (jsonConfig) {

   508→        corrected: jsonMetadata.corrected,
   509→        usedFallback: jsonMetadata.usedFallback
   510→      });

   514→      jsonMetadata = processResult.metadata;

   516→    }
   517→  }
   518→
   519→  // Save messages to session
   520→  await sessionDbManager.addMessage(session.id, {
   521→    role: 'user',
   522→    content: message
   523→  });
   524→
   525→  // Guardar la respuesta (si es JSON, guardar como string para el historial)
   526→  const responseForHistory = typeof finalResponse === 'string' 
   527→    ? finalResponse 
   528→    : JSON.stringify(finalResponse);
   529→  
   530→  await sessionDbManager.addMessage(session.id, {
   531→    role: 'assistant',
   532→    content: responseForHistory
   533→  });
   534→
   535→  return {
   536→    response: finalResponse,
   537→    sessionId: session.id,
   538→    jsonMetadata
   539→  };
   540→}

   672→  await sessionDbManager.clearMessages(session.id);

   675→    summary
   676→  };

  1445→// Main handler function
  1446→export async function handleTrigger(payload: AnyTriggerPayload): Promise<any> {
  1447→  const { mode } = payload;
  1448→
  1449→  switch (mode) {
  1450→    case 'chat':
  1451→      return handleChatTrigger(payload as ChatTriggerPayload);
  1452→    case 'resumen_sesion':
  1453→      return handleResumenSesionTrigger(payload as ResumenSesionTriggerPayload);
  1454→    case 'resumen_npc':
  1455→      return handleResumenNPCTrigger(payload as ResumenNPCTriggerPayload);
  1456→    case 'resumen_edificio':
  1457→      return handleResumenEdificioTrigger(payload as ResumenEdificioTriggerPayload);
  1458→    case 'resumen_pueblo':
  1459→      return handleResumenPuebloTrigger(payload as ResumenPuebloTriggerPayload);
  1460→    case 'resumen_mundo':
  1461→      return handleResumenMundoTrigger(payload as ResumenMundoTriggerPayload);
  1462→    case 'nuevo_lore':
  1463→      return handleNuevoLoreTrigger(payload as NuevoLoreTriggerPayload);
  1464→    default:
  1465→      throw new Error(`Unknown trigger mode: ${mode}`);
  1466→  }
  1467→}
