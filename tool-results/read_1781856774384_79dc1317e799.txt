    61→    61→/**
    62→    62→ * Extrae JSON de una respuesta del LLM
    63→    63→ * Maneja diferentes formatos: JSON puro, markdown con ```json, texto con JSON embebido
    64→    64→ */
    65→    65→function extractJsonFromResponse(response: string): string | null {
    66→    66→  // Intentar parsear directamente
    67→    67→  const trimmed = response.trim();
    68→    68→  if (trimmed.startsWith('{') || trimmed.startsWith('[')) {
    69→    69→    return trimmed;
    70→    70→  }
    71→    71→
    72→    72→  // Buscar JSON en bloques markdown ```json ... ```
    73→    73→  const jsonBlockMatch = response.match(/```(?:json)?\s*([\s\S]*?)```/);
    74→    74→  if (jsonBlockMatch) {
    75→    75→    return jsonBlockMatch[1].trim();
    76→    76→  }
    77→    77→
    78→    78→  // Buscar JSON embebido (primer { hasta último })
    79→    79→  const firstBrace = response.indexOf('{');
    80→    80→  const lastBrace = response.lastIndexOf('}');
    81→    81→  if (firstBrace !== -1 && lastBrace !== -1 && lastBrace > firstBrace) {
    82→    82→    return response.substring(firstBrace, lastBrace + 1);
    83→    83→  }
    84→    84→
    85→    85→  return null;
    86→    86→}

    88→    88→/**
    89→    89→ * Intenta parsear JSON y validar que tenga las keys requeridas
    90→    90→ */
    91→    91→function tryParseJson(jsonStr: string, requiredKeys?: string[]): { success: boolean; data?: Record<string, any>; error?: string } {
    92→    92→  try {
    93→    93→    const parsed = JSON.parse(jsonStr);
    94→    94→    
    95→    95→    // Si hay keys requeridas, validar que existan
    96→    96→    if (requiredKeys && Array.isArray(requiredKeys)) {
    97→    97→      const missingKeys = requiredKeys.filter(key => !(key in parsed));
    98→    98→      if (missingKeys.length > 0) {
    99→    99→        return { success: false, error: `Missing required keys: ${missingKeys.join(', ')}` };
   100→   100→      }
   101→   101→    }
   102→   102→    
   103→   103→    return { success: true, data: parsed };
   104→   104→  } catch (error) {
   105→   105→    return { success: false, error: error instanceof Error ? error.message : 'Unknown parse error' };
   106→   106→  }
   107→   107→}

   145→   145→

   147→   147→    console.log('[processJsonResponse] JSON parseado exitosamente en primer intento');
   148→   148→    return {
   149→   149→      success: true,
   150→   150→      data: parseResult.data!,
   151→   151→      rawResponse: response,
   152→   152→      metadata
   153→   153→    };
   154→   154→  }
   155→   155→
   156→   156→  console.log('[processJsonResponse] Error parseando JSON:', parseResult.error);
   157→   157→  
   158→   158→  // Intentar corregir
   159→   159→  return await attemptCorrection(response, jsonConfig, messages, metadata, parseResult.error || 'Parse error');

   165→   165→async function attemptCorrection(
   166→   166→  originalResponse: string,
   167→   167→  jsonConfig: JsonResponseConfig,
   168→   168→  originalMessages: ChatMessage[],
   169→   169→  metadata: any,
   170→   170→  errorMessage: string
   171→   171→): Promise<JsonProcessResult> {
   172→   172→  const maxRetries = jsonConfig.maxRetries || 2;
   173→   173→  
   174→   174→  for (let attempt = 1; attempt <= maxRetries; attempt++) {
   175→   175→    metadata.attempts = attempt + 1;
   176→   176→    console.log(`[attemptCorrection] Intento de corrección ${attempt}/${maxRetries}`);
   177→   177→    
   178→   178→    // Crear mensaje de corrección
   179→   179→    const correctionPrompt = jsonConfig.correctionPrompt || 
   180→   180→      `La respuesta anterior no tiene el formato JSON correcto. Error: ${errorMessage}
   181→   181→
   182→   182→Por favor, corrige la respuesta y devuelve ÚNICAMENTE el JSON válido sin texto adicional.
   183→   183→
   184→   184→Respuesta incorrecta:
   185→   185→${originalResponse}
   186→   186→
   187→   187→Formato esperado:
   188→   188→${JSON.stringify(jsonConfig.exampleResponse || jsonConfig.schema, null, 2)}`;
   189→   189→
   190→   190→    const correctionMessages: ChatMessage[] = [
   191→   191→      ...originalMessages,
   192→   192→      {
   193→   193→        role: 'assistant',
   194→   194→        content: originalResponse,
   195→   195→        timestamp: new Date().toISOString()
   196→   196→      },
   197→   197→      {
   198→   198→        role: 'user',
   199→   199→        content: correctionPrompt,
   200→   200→        timestamp: new Date().toISOString()
   201→   201→      }
   202→   202→    ];
   203→   203→
   204→   204→    try {
   205→   205→      const correctedResponse = await callLLM(correctionMessages);
   206→   206→      console.log('[attemptCorrection] Respuesta de corrección recibida');
   207→   207→      
   208→   208→      const jsonStr = extractJsonFromResponse(correctedResponse);
   209→   209→      if (!jsonStr) {
   210→   210→        console.log('[attemptCorrection] Aún no se puede extraer JSON');
   211→   211→        originalResponse = correctedResponse;
   212→   212→        continue;
   213→   213→      }
   214→   214→
   215→   215→      const parseResult = tryParseJson(jsonStr);
   216→   216→      if (parseResult.success) {
   217→   217→        console.log('[attemptCorrection] JSON corregido exitosamente');
   218→   218→        metadata.corrected = true;
   219→   219→        return {
   220→   220→          success: true,
   221→   221→          data: parseResult.data!,
   222→   222→          rawResponse: correctedResponse,
   223→   223→          metadata
   224→   224→        };
   225→   225→      }
   226→   226→      
   227→   227→      originalResponse = correctedResponse;
   228→   228→      errorMessage = parseResult.error || 'Parse error';
   229→   229→    } catch (error) {
   230→   230→      console.error('[attemptCorrection] Error llamando al LLM:', error);
   231→   231→    }
   232→   232→  }
   233→   233→
   234→   234→  // Agotados los intentos, usar fallback si existe
   235→   235→  if (jsonConfig.fallbackResponse) {
   236→   236→    console.log('[attemptCorrection] Usando respuesta de fallback');
   237→   237→    metadata.usedFallback = true;
   238→   238→    return {
   239→   239→      success: true,
   240→   240→      data: jsonConfig.fallbackResponse,
   241→   241→      rawResponse: originalResponse,
   242→   242→      metadata
   243→   243→    };
   244→   244→  }
   245→   245→
   246→   246→  // Sin fallback, retornar error
   247→   247→  return {
   248→   248→    success: false,
   249→   249→    data: originalResponse,
   250→   250→    rawResponse: originalResponse,
   251→   251→    metadata: {
   252→   252→      ...metadata,
   253→   253→      error: `Failed after ${maxRetries} correction attempts: ${errorMessage}`
   254→   254→    }
   255→   255→  };
   256→   256→}

   258→   258→async function callLLM(messages: ChatMessage[]): Promise<string> {
   259→   259→  try {
   260→   260→    const response = await fetch(LLM_API_URL, {
   261→   261→      method: 'POST',
   262→   262→      headers: {
   263→   263→        'Content-Type': 'application/json'
   264→   264→      },
   265→   265→      body: JSON.stringify({
   266→   266→        model: LLM_MODEL,
   267→   267→        messages: messages.map(m => ({ role: m.role, content: m.content })),
   268→   268→        temperature: LLM_TEMPERATURE,
   269→   269→        max_tokens: LLM_MAX_TOKENS
   270→   270→      })
   271→   271→    });
   272→   272→
   273→   273→    if (!response.ok) {
   274→   274→      throw new Error(`LLM API error: ${response.status} ${response.statusText}`);
   275→   275→    }
   276→   276→
   277→   277→    const data = await response.json();
   278→   278→    return data.choices[0]?.message?.content || '';
   279→   279→  } catch (error) {
   280→   280→    console.error('Error calling LLM:', error);
   281→   281→    throw error;
   282→   282→  }
   283→   283→}

   285→   285→/**
   286→   286→ * Realiza un merge incremental de datos del jugador
   287→   287→ * - Los nuevos datos sobrescriben los existentes
   288→   288→ * - Los datos que no vienen se conservan
   289→   289→ * - Los campos vacíos en el payload NO borran los existentes (solo null borra)
   290→   290→ */
   291→   291→function mergeJugadorData(
   292→   292→  jugadorExistente: Jugador | undefined,
   293→   293→  jugadorNuevo: Jugador | undefined
   294→   294→): Jugador | undefined {
   295→   295→  // Si no hay datos nuevos, conservar existentes
   296→   296→  if (!jugadorNuevo || Object.keys(jugadorNuevo).length === 0) {
   297→   297→    return jugadorExistente;
   298→   298→  }
   299→   299→
   300→   300→  // Si no hay datos existentes, usar nuevos (filtrando vacíos)
   301→   301→  if (!jugadorExistente) {
   302→   302→    const jugadorFiltrado = Object.entries(jugadorNuevo)
   303→   303→      .filter(([_, valor]) => valor !== undefined && valor !== '')
   304→   304→      .reduce((obj, [key, valor]) => ({ ...obj, [key]: valor }), {});
   305→   305→
   306→   306→    return Object.keys(jugadorFiltrado).length > 0 ? jugadorFiltrado : undefined;
   307→   307→  }
   308→   308→
   309→   309→  // Merge: nuevos sobrescriben existentes
   310→   310→  // Campos vacíos en payload NO borran, conservan valor anterior
   311→   311→  // null en payload SÍ borra explícitamente
   312→   312→  const merged = { ...jugadorExistente };
   313→   313→
   314→   314→  for (const [key, valor] of Object.entries(jugadorNuevo)) {
   315→   315→    if (valor === null) {
   316→   316→      // null significa borrar explícitamente
   317→   317→      delete (merged as any)[key];
   318→   318→    } else if (valor !== undefined && valor !== '') {
   319→   319→      // Solo actualizar si el nuevo valor no es vacío
   320→   320→      (merged as any)[key] = valor;
   321→   321→    }
   322→   322→    // Si valor es "" o undefined, conservar el existente
   323→   323→  }
   324→   324→
   325→   325→  return merged;
   326→   326→}

  1445→  1445→// Main handler function
  1446→  1446→export async function handleTrigger(payload: AnyTriggerPayload): Promise<any> {
  1447→  1447→  const { mode } = payload;
  1448→  1448→
  1449→  1449→  switch (mode) {
  1450→  1450→    case 'chat':
  1451→  1451→      return handleChatTrigger(payload as ChatTriggerPayload);
  1452→  1452→    case 'resumen_sesion':
  1453→  1453→      return handleResumenSesionTrigger(payload as ResumenSesionTriggerPayload);
  1454→  1454→    case 'resumen_npc':
  1455→  1455→      return handleResumenNPCTrigger(payload as ResumenNPCTriggerPayload);
  1456→  1456→    case 'resumen_edificio':
  1457→  1457→      return handleResumenEdificioTrigger(payload as ResumenEdificioTriggerPayload);
  1458→  1458→    case 'resumen_pueblo':
  1459→  1459→      return handleResumenPuebloTrigger(payload as ResumenPuebloTriggerPayload);
  1460→  1460→    case 'resumen_mundo':
  1461→  1461→      return handleResumenMundoTrigger(payload as ResumenMundoTriggerPayload);
  1462→  1462→    case 'nuevo_lore':
  1463→  1463→      return handleNuevoLoreTrigger(payload as NuevoLoreTriggerPayload);
  1464→  1464→    default:
  1465→  1465→      throw new Error(`Unknown trigger mode: ${mode}`);
  1466→  1466→  }
  1467→  1467→}
